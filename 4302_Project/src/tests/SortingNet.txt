class ArrayTest{
	declare{
		int[] arr = {1, 3, 4, 2};
		int temp;
		int old_i;
		int old_j;
	}
	
	func void sort(){
		swap(0,1);
		swap(2,3);
		swap(0,2);
		swap(1,3);
		swap(1,2);
		
		ensure{
			arr[0] <= arr[1];
			arr[1] <= arr[2];
			arr[2] <= arr[3];	
		}
	}
	
	func void swap(int i, int j){
		require{
			i > 0 && j > 0;
		}
		
		old_i = arr[i];
		old_j = arr[j];
		
		if (arr[i] > arr[j]) { 
			tmp = arr[i];
			arr[i] = arr[j]; 
			arr[j] = tmp; 
		}
		
		ensure{
			(old_i > old_j) => (arr[i] == old_j) && (arr[j] == old_i);
		}
	}
	
}
