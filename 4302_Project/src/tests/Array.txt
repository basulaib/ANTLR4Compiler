class ArrayTest{
	declare{
		int[] arr1 = {1, 3, 4, 2};
		int[2] arr2;
		int temp;
		int oldi;
		int oldj;
	}
	
	func void swap(int i, int j){
		//x == 1 means increasing, x == 0 means decreasing order
		require{
			i > 0 && j > 0 && i < 4 && j < 4;
		}
		
		temp = arr1[1];
		arr1[1] = arr1[2];
		arr1[2] = temp;
		
		oldi = arr1[i];
		oldj = arr1[j];
		
		temp = arr1[i];
		arr1[i] = arr1[j];
		arr1[j] = temp;
		
		ensure{
			arr1[i] == oldj;
			arr1[j] == oldi;
			arr1[1] == 4;
			arr1[2] == 3;
		}
	}
	
}
