class Sorting{
	declare{
		int a = 6;
		int b = 1;
		int c = 4; //if you change this to 8, the verification will also fail because <= is not in post-condition!
		int d = 8;
		int tmp;
	}
	
	func void sort(int x){
		//x == 1 means increasing, x == 0 means decreasing order
		require{
			x == 1 || x == 0;	
		}
		
		if (x == 1){
			if (a > b) { tmp = a; a = b; b = tmp; }
			if (c > d) { tmp = c; c = d; d = tmp; }
			if (a > c) { tmp = a; a = c; c = tmp; }
			if (b > d) { tmp = b; b = d; d = tmp; }
			if (b > c) { tmp = b; b = c; c = tmp; }
		}else{
			if (a < b) { tmp = a; a = b; b = tmp; }
			if (c < d) { tmp = c; c = d; d = tmp; }
			if (a < c) { tmp = a; a = c; c = tmp; }
			if (b < d) { tmp = b; b = d; d = tmp; }
			if (b < c) { tmp = b; b = c; c = tmp; }
		}
		
		ensure{
			x == 1 => (a < b && b < c && c < d); //if you change c < d to c > d, program becomes invalid
			x == 0 => (a > b && b > c && c > d);
		}
	}
	
}
