class Professor {
    declare {
        string name = "Trudeau";
        string course = "EECS 4302";
        int pay = 95000;
        bool tenure = false;
        int exp_years = 1;
    }
    
    assume {
        pay > 50000;
        tenure == false;
    }

    func checkTenureTrack() {
        require {
            exp_years > 0;
            tenure == false;
        }

        if (exp_years > 10) {
            tenure = true;
        }
    }

    assert {
        pay > 50000;
        exp_years >= 1;
    }
}

class Course {
    declare {
        string name;
        int code;
        int fee;
        string semester;
        int year;
        string instructor;
        string department;
        string lecture_room;
    }

    assume {
        name != "";
        code > 0;
        fee > 0;
        (semester == "F") || (semester == "W") || (semester == "S1") || (semester == "S2") || (semester == "SU");
        (year > 0) && (year < 5);
    }
}
