//test for conjunction/equivalence
class test{
	declare {
    	int result;
		int a = 5;
		int amount;
    }

	assume{
		amount > 0;
	}

//   amount = 5;
//   ID = 8;

	func int mult(int x, int y){
		require {
			x >= 0;
			y >= 0;
		}

		result = x * y;

		ensure {
			result == x * y;
		}
	}

	assert {
		result >= 0;
	}
}
