class test{
	declare {
    	int result;
    	int amount;
    }


	func int negate(int y){
		require {
			y > 0;
		}

		amount = 5;
		
		switch (amount){
			case: 3 {
				result = 4;
			}
			
			case: 5 {
				result = 7;
			}
			
			default {
				result = 5;
			}
		};

		ensure {
			result <=> 7;
		}
	}

	assert {
		amount <=> 5;
	}
}
